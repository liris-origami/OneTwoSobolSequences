export_matrices:
=

reads a file (Joe&Kuo format), build matrices using Sobol' recurrence and export matrices as C-style arrays with efficient point generation functions.

usage:
```
	# export Joe&Kuo matrices
	./export_matrices data/sobol_init_tab.dat sobol	
	# export OneTwo matrices
	./export_matrices data/onetwo_init_tab.dat onetwo
```


example:
==
```
	# export Joe&Kuo matrices
	./export_matrices data/sobol_init_tab.dat sobol	
```

- sobol_matrices.h file:
```
#pragma once

#include <cassert>
#include <cstdint>
#include <array>

inline
unsigned sobol_sample( const std::array<unsigned, 52>& m, uint64_t n )
{
    unsigned x= 0;
    for(unsigned i= 0; i < 52; i++)
        x ^= ((n >> i) & 1) * m[i];
    
    return x;
}

constexpr unsigned sobol_matrices_size= 1024;
extern std::array<unsigned, 52> sobol_matrices[sobol_matrices_size];

inline
unsigned sobol_sample( const unsigned d, const unsigned index )
{
    assert(d < sobol_matrices_size);
    return sobol_sample(sobol_matrices[d], index);
}
```
